解题思路：
    先排序，之后每次贪心选择第二大，所以一次选择n个第二大值累加即可
```cpp
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
    int n;
    while (cin >> n)
        {
            long long sum = 0;
            vector<int> a; a.resize(3*n);
            for (int i = 0; i < (3 * n); i++) {
                cin >> a[i];
            }
    std::sort(a.begin(), a.end());
    for(int i = n; i <= 3 * n - 2; i += 2)
        {
            sum += a[i];
        }
    cout << sum << endl;
    }
}
```
